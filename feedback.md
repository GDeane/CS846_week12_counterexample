"""
Task Queue with Dispatcher and Worker Threads.

A dispatcher assigns tasks from a shared queue to a pool of worker threads.
Each worker must acquire a shared resource lock before processing its task.

Known bug: ~10% of tasks trigger a simulated deadlock inside perform_computation,
causing that worker thread to hang indefinitely without raising an exception.
The thread simply stops making progress while holding or waiting on the lock.
"""

import threading
import time
import random
import logging
from queue import Queue, Empty


def perform_computation(task_id: int) -> str:
    """Simulate CPU-bound work. Occasionally blocks forever (deadlock simulation)."""
    if random.random() < 0.10:
        # Simulates a thread deadlocking: waits on an Event that is never set.
        threading.Event().wait()
    time.sleep(random.uniform(0.05, 0.2))
    return f"result_for_{task_id}"


def worker_thread(
    worker_id: int,
    task_queue: Queue,
    resource_lock: threading.Lock,
    results: dict,
    stop_event: threading.Event,
) -> None:
    """
    Pull tasks from the queue, acquire a shared resource lock,
    perform computation, and store the result.
    """
    logger = logging.getLogger(__name__)
    while not stop_event.is_set():
        try:
            task_id = task_queue.get(timeout=0.5)
        except Empty:
            continue

        try:
            with resource_lock:
                result = perform_computation(task_id)
        except Exception as exc:
            # Recoverable failure during task processing.
            logger.warning("Task %d failed with recoverable error: %s", task_id, exc)
            task_queue.task_done()
            continue

        # Final outcome for the task (one INFO per successful task).
        results[task_id] = result
        logger.info("Task %d completed: %s", task_id, result)
        task_queue.task_done()


def run_dispatcher(num_tasks: int, num_workers: int, timeout: float = 10.0) -> dict:
    """
    Dispatch `num_tasks` tasks across `num_workers` worker threads.
    Returns a dict mapping task_id -> result for all completed tasks.
    Blocks until all tasks finish or `timeout` seconds elapse.
    """
    task_queue: Queue = Queue()
    resource_lock = threading.Lock()
    results: dict = {}
    stop_event = threading.Event()

    for task_id in range(num_tasks):
        task_queue.put(task_id)

    workers = []
    for worker_id in range(num_workers):
        t = threading.Thread(
            target=worker_thread,
            args=(worker_id, task_queue, resource_lock, results, stop_event),
            daemon=True,
        )
        t.start()
        workers.append(t)

    finished = threading.Event()

    def wait_for_queue():
        task_queue.join()
        finished.set()

    threading.Thread(target=wait_for_queue, daemon=True).start()
    # If no logging configuration exists, enable a sensible default so
    # diagnostic messages are visible in simple runs.
    if not logging.getLogger().handlers:
        logging.basicConfig(level=logging.INFO)

    finished.wait(timeout=timeout)

    logger = logging.getLogger(__name__)
    if not finished.is_set():
        logger.error("Timeout reached before all tasks completed (timeout=%.1f seconds)", timeout)

    stop_event.set()
    for t in workers:
        t.join(timeout=1.0)

    # Any tasks still in the queue after the timeout are treated as errors.
    while True:
        try:
            task_id = task_queue.get_nowait()
        except Empty:
            break
        logger.error("Task %d could not be completed (dispatcher timeout)", task_id)
        task_queue.task_done()

    return results
```

### 3. New and Updated Guidelines that worked for Problem A_1

#### Guideline 6 (Revised): [Use a context-aware heartbeat budget for concurrent systems]

**Description:**

When adding logging to a multi-threaded or asynchronous system, the logging budget must
account for **state transitions**, not just final outcomes. Require exactly two INFO logs
per unit of work: one when the work is **dequeued** (start) and one when it
**completes** (end). A unit of work that shows a start entry but no end entry
is immediately recognisable as hung. Allow WARNING for recoverable errors and ERROR for
terminal failures, as in the original guideline.

**Reasoning:**

The original Guideline 6 is well-suited to synchronous pipelines where every started
operation eventually reaches a final outcome. In concurrent systems, however, a thread
can stop making progress without raising an exception — it simply never reaches the
outcome log. The INFO pair (start + end) acts as a liveness signal: the *absence*
of the second log is itself diagnostic information. The two-log-per-task budget is still
modest and avoids the "per-record noise" the original guideline warns against.

**Prompt and Context:**

Context Given: task_queue.py

> Add logging to this multi-threaded task dispatcher.
> Rules:
> - Exactly 2 INFO logs per task: one when the task is **dequeued** (include worker_id
    > and task_id) and one when it **completes** (include worker_id, task_id, and result).
    > These heartbeat logs are required to detect hung threads in production.
> - Use WARNING only when a task fails with a recoverable error.
> - Use ERROR only when a task cannot be completed.
> - Do not log per-worker loop iterations or internal helper calls.

"""
Task Queue with Dispatcher and Worker Threads.

A dispatcher assigns tasks from a shared queue to a pool of worker threads.
Each worker must acquire a shared resource lock before processing its task.

Known bug: ~10% of tasks trigger a simulated deadlock inside perform_computation,
causing that worker thread to hang indefinitely without raising an exception.
The thread simply stops making progress while holding or waiting on the lock.
"""

import threading
import time
import random
import logging
from queue import Queue, Empty

# Configure logging to ensure INFO heartbeats are emitted when module is run directly.
# Only configure the root logger if no handlers are present to avoid clobbering app config.
if not logging.getLogger().hasHandlers():
    logging.basicConfig(level=logging.INFO)


def perform_computation(task_id: int) -> str:
    """Simulate CPU-bound work. Occasionally blocks forever (deadlock simulation)."""
    if random.random() < 0.10:
        # Simulates a thread deadlocking: waits on an Event that is never set.
        threading.Event().wait()
    time.sleep(random.uniform(0.05, 0.2))
    return f"result_for_{task_id}"


def worker_thread(
    worker_id: int,
    task_queue: Queue,
    resource_lock: threading.Lock,
    results: dict,
    stop_event: threading.Event,
) -> None:
    """
    Pull tasks from the queue, acquire a shared resource lock,
    perform computation, and store the result.
    """
    logger = logging.getLogger(__name__)

    while not stop_event.is_set():
        try:
            task_id = task_queue.get(timeout=0.5)
        except Empty:
            continue

        # Heartbeat: task dequeued (INFO) — required per task
        logger.info("Task dequeued: worker=%s task=%s", worker_id, task_id)

        result = None
        try:
            with resource_lock:
                result = perform_computation(task_id)
        except Exception as e:
            # If an exception indicates a recoverable condition, log WARNING;
            # otherwise log ERROR. We preserve an INFO 'completed' log below
            # so monitoring always sees exactly two INFOs per task.
            if getattr(e, "recoverable", False):
                logger.warning(
                    "Recoverable error processing task: worker=%s task=%s error=%s",
                    worker_id,
                    task_id,
                    e,
                )
            else:
                logger.error(
                    "Failed to process task: worker=%s task=%s error=%s",
                    worker_id,
                    task_id,
                    e,
                )
        else:
            # Only store successful results
            results[task_id] = result
        finally:
            # Heartbeat: task completed (INFO) — required per task
            logger.info(
                "Task completed: worker=%s task=%s result=%s",
                worker_id,
                task_id,
                result,
            )
            task_queue.task_done()


def run_dispatcher(num_tasks: int, num_workers: int, timeout: float = 10.0) -> dict:
    """
    Dispatch `num_tasks` tasks across `num_workers` worker threads.
    Returns a dict mapping task_id -> result for all completed tasks.
    Blocks until all tasks finish or `timeout` seconds elapse.
    """
    task_queue: Queue = Queue()
    resource_lock = threading.Lock()
    results: dict = {}
    stop_event = threading.Event()

    for task_id in range(num_tasks):
        task_queue.put(task_id)

    workers = []
    for worker_id in range(num_workers):
        t = threading.Thread(
            target=worker_thread,
            args=(worker_id, task_queue, resource_lock, results, stop_event),
            daemon=True,
        )
        t.start()
        workers.append(t)

    finished = threading.Event()

    def wait_for_queue():
        task_queue.join()
        finished.set()

    threading.Thread(target=wait_for_queue, daemon=True).start()
    finished.wait(timeout=timeout)

    stop_event.set()
    for t in workers:
        t.join(timeout=1.0)

    return results
```

def _is_blacklisted(
	rules: list[BlacklistRule],
	*,
	txn_type: str,
	amount: Decimal,
	currency: str,
	from_acct: Optional[str],
	to_acct: Optional[str],
) -> bool:
	"""Return True if any rule blocks this transaction (silent skip)."""
	i = 0
	while i < len(rules):
		r = rules[i]
		i += 1

		if r.txn_type != txn_type:
			continue
		if r.currency is not None and r.currency != currency:
			continue
		if r.min_amount is not None and amount < r.min_amount:
			continue
		if r.max_amount is not None and amount > r.max_amount:
			continue

		# Which side does the rule apply to?

		counterparty = to_acct if r.applies_to == "from" else from_acct

			continue
		if subject != r.account_id:
			continue
		if r.counterparty_account is not None and counterparty != r.counterparty_account:
			continue

		return True

	return False
